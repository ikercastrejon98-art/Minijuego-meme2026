<script>

let secret, lives, user;

function startGame() {
    user = document.getElementById("name").value.trim();
    if (!user) return alert("Pon nombre bro");

    secret = Math.floor(Math.random() * 10) + 1;
    lives = 3;

    document.getElementById("game").style.display = "block";
    document.getElementById("end").style.display = "none";
    document.getElementById("msg").innerText = "";
    document.getElementById("meme").style.display = "none";

    updateLives();
}

function updateLives() {
    document.getElementById("lives").innerText = "❤️".repeat(lives);
}

function showMeme(type) {
    const meme = document.getElementById("meme");
    meme.style.display = "block";

    const memes = {
        win: "https://i.imgur.com/3ZQ3Z6F.gif",
        lose: "https://i.imgur.com/j6qY6Qp.gif"
    };

    meme.src = memes[type];
}

function check() {
    let guess = Number(document.getElementById("guess").value);
    let msg = document.getElementById("msg");

    if (isNaN(guess)) {
        msg.innerText = "⚠️ Número inválido";
        return;
    }

    if (guess === secret) {
        win();
    } else {
        lives--;
        updateLives();

        if (lives <= 0) {
            lose();
        } else if (guess < secret) {
            msg.innerText = "📉 Muy bajo";
        } else {
            msg.innerText = "📈 Muy alto";
        }
    }

    document.getElementById("guess").value = "";
}

function win() {
    document.getElementById("result").innerText = "🔥 GANASTE";
    showMeme("win");
    saveScore(100);
    endGame();
}

function lose() {
    document.getElementById("result").innerText = "💀 Perdiste. Era " + secret;
    showMeme("lose");
    endGame();
}

function endGame() {
    document.getElementById("game").style.display = "none";
    document.getElementById("end").style.display = "block";
}

function saveScore(points) {
    let data = JSON.parse(localStorage.getItem("rank")) || {};
    data[user] = (data[user] || 0) + points;
    localStorage.setItem("rank", JSON.stringify(data));
    showRanking();
}

function showRanking() {
    let data = JSON.parse(localStorage.getItem("rank")) || {};
    let list = document.getElementById("ranking");

    list.innerHTML = "";

    Object.entries(data)
    .sort((a,b)=>b[1]-a[1])
    .forEach((p,i) => {
        list.innerHTML += `<li>#${i+1} ${p[0]} - ${p[1]} pts</li>`;
    });
}

showRanking();
