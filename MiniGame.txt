<script>

let secret, lives, user;

function startGame() {
    user = document.getElementById("name").value.trim();
    if (!user) return alert("Pon nombre 💀");

    secret = Math.floor(Math.random() * 10) + 1;
    lives = 3;

    document.getElementById("game").style.display = "block";
    document.getElementById("end").style.display = "none";
    document.getElementById("msg").innerText = "";
    document.getElementById("meme").style.display = "none";

    updateLives();
}

function updateLives() {
    document.getElementById("lives").innerText = "❤️".repeat(lives);
}

function randomMeme(type) {
    const memes = {
        win: [
            "https://i.imgur.com/3ZQ3Z6F.gif",
            "https://i.imgur.com/6X4KZQW.gif"
        ],
        lose: [
            "https://i.imgur.com/j6qY6Qp.gif",
            "https://i.imgur.com/Z4M7K6B.gif"
        ],
        high: [
            "https://i.imgur.com/FW5RZ8C.gif"
        ],
        low: [
            "https://i.imgur.com/8p0XK4R.gif"
        ]
    };

    let list = memes[type];
    return list[Math.floor(Math.random() * list.length)];
}

function showMeme(type) {
    const meme = document.getElementById("meme");
    meme.style.display = "block";
    meme.src = randomMeme(type);
}

function check() {
    let guess = Number(document.getElementById("guess").value);
    let msg = document.getElementById("msg");

    if (isNaN(guess)) {
        msg.innerText = "💀 ni escribir sabes";
        return;
    }

    if (guess === secret) {
        win();
    } else {
        lives--;
        updateLives();

        if (lives <= 0) {
            lose();
        } else if (guess < secret) {
            msg.innerText = "📉 Súbele crack";
            showMeme("low");
        } else {
            msg.innerText = "📈 bájale tantito";
            showMeme("high");
        }
    }

    document.getElementById("guess").value = "";
}

function win() {
    document.getElementById("result").innerText = "🔥 ERES DIOS";
    document.body.classList.add("win");
    showMeme("win");
    playSound("win");
    saveScore(100);
    endGame();
}

function lose() {
    document.getElementById("result").innerText = "💀 NPC DETECTED";
    showMeme("lose");
    playSound("lose");
    endGame();
}

function endGame() {
    document.getElementById("game").style.display = "none";
    document.getElementById("end").style.display = "block";
}

function saveScore(points) {
    let data = JSON.parse(localStorage.getItem("rank")) || {};
    data[user] = (data[user] || 0) + points;
    localStorage.setItem("rank", JSON.stringify(data));
    showRanking();
}

function showRanking() {
    let data = JSON.parse(localStorage.getItem("rank")) || {};
    let list = document.getElementById("ranking");

    list.innerHTML = "";

    Object.entries(data)
    .sort((a,b)=>b[1]-a[1])
    .slice(0,3)
    .forEach((p,i) => {
        let medal = ["🥇","🥈","🥉"][i] || "";
        list.innerHTML += `<li>${medal} ${p[0]} - ${p[1]}</li>`;
    });
}

function playSound(type) {
    let audio = new Audio(
        type === "win"
        ? "https://www.myinstants.com/media/sounds/victory.mp3"
        : "https://www.myinstants.com/media/sounds/fail-trombone.mp3"
    );
    audio.play();
}

showRanking();
